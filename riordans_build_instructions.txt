editor
------
scons platform=windows tools=yes target=editor -j10 custom_modules=./custom_modules

templates
---------
scons platform=windows target=template_debug arch=x86_64 custom_modules=./custom_modules -j10 profile=custom.py
scons platform=web target=template_debug arch=x86_64 custom_modules=./custom_modules -j10 profile=custom.py


VS Project
----------
scons platform=windows tools=yes target=editor -j10 custom_modules=./custom_modules vsproj=yes debug_symbols=true
